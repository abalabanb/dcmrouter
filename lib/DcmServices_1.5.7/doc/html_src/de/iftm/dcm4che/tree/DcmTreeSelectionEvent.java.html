package de.iftm.dcm4che.tree;

import java.util.*;

import org.apache.log4j.*;


/**
 * @author Thomas Hacklaender
 * @version 2006.06.07
 */
public class DcmTreeSelectionEvent extends java.util.EventObject {
    
    /** Initialize logger */
    private static Logger log = Logger.getLogger(DcmTreeSelectionEvent.class);

        
    private Vector selectedFiles = null;
 
    
    /**
     * Creates a new instance of DcmTreeSelectionEvent
     * 
     * @param source
     */
    public DcmTreeSelectionEvent(Object source) {
        // pass the source object to the supercalss
        super(source);
    }

    
    /**
     * Creates a new instance of DcmTreeSelectionEvent
     * 
     * @param source
     * @param selectedFiles
     */
    public DcmTreeSelectionEvent(Object source, Vector selectedFiles) {
        // pass the source object to the supercalss
        super(source);
        setSelectedFiles(selectedFiles);
    }
    
    
    /**
     *
     * @param selectedFiles
     */
    public void setSelectedFiles(Vector selectedFiles) {
        this.selectedFiles = selectedFiles;
    }
    
    
    /**
     * 
     * @return 
     */
    public Vector getSelectedFiles() {
        return selectedFiles;
    }
 
}

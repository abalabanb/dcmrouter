    /**

    public void insertDICOMDIR(File f) throws IOException {
        DirReader dirReader = null;
        DirRecord dr;
        Dataset ds;
        DcmObjectRecord dor;
        
        // The content of the directory records of a DICOMDIR are defined
        // in PS 3.3 - F.5 DEFINITION OF SPECIFIC DIRECTORY RECORDS.
        //
        // List of requieren Type 1 and 2 items:
        //
        // F.5.1 Patient Directory Record Definition
        //   - Specific Character Set, Type 1C
        //   - Patient's Name, Type 2
        //   - Patient ID, Type 1
        //
        // F.5.2 Study Directory record definition
        //   - Specific Character Set, Type 1C
        //   - Study Date, Type 1
        //   - Study Time, Type 1
        //   - Study Description, Type 2
        //   - Study Instance UID, Type 1C: The Study Instance UID shall be present as a mandatory key only if no file is referenced by this Directory Record.
        //   - Study ID, Type 1
        //   - Accession Number, Type 2
        //
        // F.5.3 Series Directory Record Definition
        //   - Specific Character Set, Type 1C
        //   - Modality, Type 1
        //   - Series Instance UID, Type 1
        //   - Series Number, Type 1
        //
        // F.5.4 Image directory record definition
        //   - Specific Character Set, Type 1C
        //   - Instance Number, Type 1
        
        // Get a DirReader
        dirReader = DirBuilderFactory.getInstance().newDirReader(f);
        
        // Get first Patient
        dr = dirReader.getFirstRecordBy(DirRecord.PATIENT, null, false);
        
        // Loop for all patients
        while (dr != null) {
            
            // Use PATIENT DirRecord as inital value for DcmObjectRecord
            dor = new DcmObjectRecord(f);
            dor.addPatientModule(dr.getDataset());
            
            try {
                getNextChild(dr, dor);
            } catch (IOException e) {
                log.error(e.getMessage());
            }
            
            // Get next Patient
            dr = dr.getNextSibling();
        }
        
    }
    
    
    private void getNextChild(DirRecord dr, DcmObjectRecord dor) throws IOException {
        DirRecord child;
        
        child = dr.getFirstChild();
        
        // Loop for all childs
        while (child != null) {
            
            // Hanging Protocol DR and HL7 Struc Doc DR
            // are defined in PS 3.3 - F.5. but not in dcm4che 1.3.x
            // Should be catched before processing other types
            
            if (child.getType().equals(DirRecord.PATIENT)) {
                
                // PATIENT
                dor.addPatientModule(child.getDataset());
                getNextChild(child, dor);
                
            } else if (child.getType().equals(DirRecord.STUDY)) {
                
                // STUDY
                dor.addStudyModule(child.getDataset());
                getNextChild(child, dor);
                
            } else if (child.getType().equals(DirRecord.SERIES)) {
                // SERIES
                dor.addSeriesModule(child.getDataset());
                getNextChild(child, dor);
                
            } else {
                
                // Allow all type of instance
                dor.addInstanceModule(child.getDataset());
                if (child.getRefSOPClassUID() != null) {
                    dor.sopClassUID = child.getRefSOPClassUID();
                }
                if (child.getRefSOPInstanceUID() != null) {
                    dor.sopClassUID = child.getRefSOPInstanceUID();
                }
                
                String[] sa = child.getRefFileIDs();
                
                // Insert to database only if file IDs are available
                if (!(sa == null)) {
                    dor.buildFilePath(sa);
                    
//                    log.info(">>>>>>>>>>>>>>>>");
//                    log.info(dor.patientName);
//                    log.info(dor.studyID);
//                    log.info(dor.studyDescription);
//                    log.info(dor.studyInstanceUID);
//                    log.info(dor.seriesNumber);
//                    log.info(dor.seriesInstanceUID);
//                    log.info(dor.instanceNumber);
//                    log.info(dor.sopInstanceUID);
//                    log.info(dor.sopClassUID);
//                    log.info(dor.filePath);
                    
                    // Insert referenced file to database
                    try {
                        insertDcmObjectRecord(dor);
                    } catch (SQLException e) {
                        log.error(e.getMessage());
                    }
                }
            }
            
            child = child.getNextSibling();
        }
    }
    
    
    /**
     *
     */
    public void insertFiles(File f, boolean deep) {
        String sql;
        File[] contentFiles;
        
        // If input is empty, nothing to do
        if (f == null) {
            return;
        }
        
        // If input is a file, just add it
        if (! f.isDirectory()) {
            try {
                insertSingleDcmFile(f);
                return;
            } catch (SQLException e) {
                log.error(e.getMessage());
            }
        }
        
        // Input is a directory
        
        contentFiles = f.listFiles();
        
        // If no content, nothing to do
        if (contentFiles == null) {
            return;
        }
        
        for (int i=0; i<contentFiles.length; i++) {
            
            if (contentFiles[i].isDirectory()) {
                
                // Content item is a directory
                if (deep) {
                    // Process recursively
                    insertFiles(contentFiles[i], true);
                } else {
                    // Ignore
                }
                
            } else {
                
                // Content item is a file, add it
                try {
                    insertSingleDcmFile(contentFiles[i]);
                } catch (SQLException e) {
                    log.error(e.getMessage());
                }
                
            }
        }
    }
    
    
    /**
     *
     */
    private synchronized void insertSingleDcmFile(File f) throws SQLException {
        Dataset ds;
        DcmObjectRecord dor;
        
        
        // Return if now file given
        if (f == null) {
            return;
        }
        
        // Return if directory
        if (f.isDirectory()) {
            return;
        }
        
        // Return if file is already in database
        if (isFileInDatabase(f)) {
            return;
        }
        
        // Return if not a DICOM file
        ds = fileToDataset(f);
        if (ds == null) {
            return;
        }
        
        // New header obeject
        dor = new DcmObjectRecord(f, ds);
        
        // Store it in database
        insertDcmObjectRecord(dor);
    }
    
    
    /**
     *
     */
    private synchronized void insertDcmObjectRecord(DcmObjectRecord dor) throws SQLException {
        
        // Ignore records without SOP Class UID
        if (dor.sopClassUID.length() == 0) {
            log.warn("No SOP Class UID. File not included: " + dor.filePath);
            return;
        }
        
        PreparedStatement ps = conn.prepareStatement("INSERT INTO dcm_table(" +
                "isProcessed, " +
                "filePath, " +
                "patientName, " +
                "patientID, " +
                "patientBirthDate, " +
                "patientSex, " +
                "studyInstanceUID, " +
                "studyDate, " +
                "studyTime, " +
                "referringPhysicianName, " +
                "studyID, " +
                "accessionNumber, " +
                "studyDescription, " +
                "modality, " +
                "seriesInstanceUID, " +
                "seriesNumber, " +
                "instanceNumber, " +
                "sopClassUID, " +
                "sopInstanceUID " +
                ") VALUES(?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)");
        
        // Item is not processed
        ps.setBoolean(1, false);
        
        ps.setString(2, dor.filePath);
        
        // Patient IE, Patient Module, PS 3.3 - C.7.1.1, M
        ps.setString(3, dor.patientName);            // Type 2
        ps.setString(4, dor.patientID);              // Type 2
        ps.setString(5, dor.patientBirthDate);       // Type 2
        ps.setString(6, dor.patientSex);             // Type 2
        
        // Study IE, General Study Module, PS 3.3 - C.7.2.1, M
        ps.setString(7, dor.studyInstanceUID);       // Type 1
        ps.setString(8, dor.studyDate);              // Type 2
        ps.setString(9, dor.studyTime);              // Type 2
        ps.setString(10, dor.referringPhysicianName); // Type 2
        ps.setString(11, dor.studyID);               // Type 2
        ps.setString(12, dor.accessionNumber);       // Type 2
        ps.setString(13, dor.studyDescription);      // Type 3
        
        // Series IE, General Series Module, PS 3.3 - C.7.3.1, M
        ps.setString(14, dor.modality);               // Type 1
        ps.setString(15, dor.seriesInstanceUID);      // Type 1
        ps.setString(16, dor.seriesNumber);           // Type 2
        
        // Image General Module, PS 3.3 - C.7.6.1 General Image Module
        ps.setString(17, dor.instanceNumber);         // Type 2
        
        // SOP Common Module, PS 3.3 - C.12.1, M
        ps.setString(18, dor.sopClassUID);            // Type 1
        ps.setString(19, dor.sopInstanceUID);         // Type 1
        
        ps.executeUpdate();
        
        ps.close();
    }
    
    
    /**
     *
     */
    private synchronized boolean isFileInDatabase(File f) throws SQLException {
        ResultSet rs;
        boolean   result;
        
        PreparedStatement ps = conn.prepareStatement("SELECT * FROM dcm_table WHERE filePath = ?");
        
        ps.setString(1, f.getPath());
        
        // run the query
        rs = ps.executeQuery();
        if (rs.next()) {
            result = true;
            // log.info("File IN database :" + f.getPath());
        } else{
            result = false;
            // log.info("File NOT in database :" + f.getPath());
        }
        
        ps.close();
        
        return result;
    }

    /**
     * Use for SQL command SELECT.
     * <p>Samples:
     * <p>  SELECT * FROM dcm_table WHERE patientName >= 'A'
     * <p>  SELECT * FROM dcm_table
     * <p>  SELECT DISTINCT patientName FROM dcm_table
     * <p>  SELECT DISTINCT studyID, studyDescription FROM dcm_table WHERE patientName >= 'A'
     */
    public synchronized void query(String expression) throws SQLException {
        
        Statement stmt = null;
        ResultSet rs = null;
        
        // Statement objects can be reused with repeated calls to execute but
        // we choose to make a new one each time
        stmt = conn.createStatement();
        
        // Run the query
        rs = stmt.executeQuery(expression);
        
        dump(rs);
        
        // NOTE!! if you close a statement the associated ResultSet is closed too.
        // So you should copy the contents to some other object.
        // The result set is invalidated also, if you recycle an Statement
        // and try to execute some other query before the result set has been
        // completely examined.
        stmt.close();
    }
    
    
    /**
     *
     */
    private static void dump(ResultSet rs) throws SQLException {
        ResultSetMetaData meta = null;
        int               colmax;
        Object            o = null;
        
        // The order of the rows in a cursor are implementation dependent unless
        // you use the SQL ORDER statement
        
        // Get an object that can be used to get information about the types and
        // properties of the columns in a ResultSet object
        meta   = rs.getMetaData();
        
        // Returns the number of columns in this ResultSet object
        colmax = meta.getColumnCount();
        
        // The result set is a cursor into the data.  You can only point to one
        // row at a time. Assume we are pointing to BEFORE the first row rs.next()
        // points to next row and returns true or false if there is no next row,
        // which breaks the loop
        while (rs.next()) {
            
            // Process row
            for (int i = 0; i < colmax; ++i) {
                
                // Is SQL the first column is indexed with 1 not 0
                o = rs.getObject(i + 1);
                
                // Output the object
                log.info(o.toString() + " ");
            }
            
            // Output empty line
            log.info("<<<< end row <<<<<");
        }
    }
    
    
    /**
     * Loads a Dataset object from a file.
     * @param the filr containing teh Dataset.
     * @result the read in Dataset.
     */
    private Dataset fileToDataset(File f) {
        BufferedInputStream in = null;
        FileFormat          ff = null;
        DcmParser           parser = null;
        Dataset             ds = null;
        byte[]              dt = new byte[4];
        
        // DICOM files must have a length of at least 132 byte
        // (128 byte header and 'DICM' token)
        if (f.length() < 132) {
            return null;
        }
        
        // Test DICOM token
        try {
            FileInputStream fis = new FileInputStream(f);
            fis.skip(128);
            fis.read(dt, 0, 4);
            fis.close();
        } catch (IOException e) {
            log.error("Can't read DICOM token: " + f.getPath());
            return null;
        }
        if (dt[0] != 'D' || dt[1] != 'I' || dt[2] != 'C' || dt[3] != 'M') {
            return null;
        }
        
        // Read DICOM file
        try {
            in = new BufferedInputStream(new FileInputStream(f));
        } catch (IOException e) {
            log.error("Can't read file: " + f.getPath());
            return null;
        }
        
        try {
            parser = DcmParserFactory.getInstance().newDcmParser(in);
            
            try {
                ff = parser.detectFileFormat();
            } catch (Exception ioe) {
                log.error("Can't detect DICOM file format for file: " + f.getPath());
                // Try next file
                return null;
            }
            
            ds = DcmObjectFactory.getInstance().newDataset();
            
            try {
                ds.readFile(in, ff, -1);
            } catch (IOException ioe) {
                log.error("Can't create Dataset for file: " + f.getPath());
                // Try next file
                return null;
            }
        } finally {
            try {
                in.close();
            } catch (IOException e) {}
        }
        
        return ds;
    }

    public class DcmObjectRecord {
        
        private File dicomdir = null;
        
        public int recordID = -1;
        public String filePath = "";
        public boolean isProcessed = false;
        
        // Patient IE, Patient Module, PS 3.3 - C.7.1.1, M
        public String patientName = "";             // Type 2
        public String patientBirthDate = "";        // Type 2
        public String patientSex = "";              // Type 2
        public String patientID = "";               // Type 2
        
        // Study IE, General Study Module, PS 3.3 - C.7.2.1, M
        public String studyInstanceUID = "";        // Type 1
        public String studyDate = "";               // Type 2
        public String studyTime = "";               // Type 2
        public String referringPhysicianName = "";  // Type 2
        public String studyID = "";                 // Type 2
        public String accessionNumber = "";         // Type 2
        public String studyDescription = "";        // Type 3
        
        // Series IE, General Series Module, PS 3.3 - C.7.3.1, M
        public String modality = "";                // Type 1
        public String seriesInstanceUID = "";       // Type 1
        public String seriesNumber = "";            // Type 2
        
        // Image General Module, PS 3.3 - C.7.6.1 General Image Module
        public String instanceNumber = "";          // Type 2
        
        // SOP Common Module, PS 3.3 - C.12.1, M
        public String sopClassUID = "";             // Type 1
        public String sopInstanceUID = "";          // Type 1
        
        
        public DcmObjectRecord(File dicomdir) {
            this.dicomdir = dicomdir;
        }
        
        
        public DcmObjectRecord(File f, Dataset ds) {
            this(ds);
            filePath = f.getPath();
        }
        
        
        public DcmObjectRecord(Dataset ds) {
            addPatientModule(ds);
            addStudyModule(ds);
            addSeriesModule(ds);
            addInstanceModule(ds);
        }
        
        
        public void addPatientModule(Dataset ds) {
            
            // Patient IE, Patient Module, PS 3.3 - C.7.1.1, M
            patientName = ds.getString(Tags.PatientName, "");                       // Type 2
            patientID = ds.getString(Tags.PatientID, "");                           // Type 2
            patientBirthDate = ds.getString(Tags.PatientBirthDate, "");             // Type 2
            patientSex = ds.getString(Tags.PatientSex, "");                         // Type 2
        }
        
        
        public void addStudyModule(Dataset ds) {
            
            // Study IE, General Study Module, PS 3.3 - C.7.2.1, M
            studyInstanceUID = ds.getString(Tags.StudyInstanceUID, "");             // Type 1
            studyDate = ds.getString(Tags.StudyDate, "");                           // Type 2
            studyTime = ds.getString(Tags.StudyTime, "");                           // Type 2
            referringPhysicianName = ds.getString(Tags.ReferringPhysicianName, ""); // Type 2
            studyID = ds.getString(Tags.StudyID, "");                               // Type 2
            accessionNumber = ds.getString(Tags.AccessionNumber, "");               // Type 2
            studyDescription = ds.getString(Tags.StudyDescription, "");             // Type 3
        }
        
        
        public void addSeriesModule(Dataset ds) {
            
            // Series IE, General Series Module, PS 3.3 - C.7.3.1, M
            modality = ds.getString(Tags.Modality, "");                             // Type 1
            seriesInstanceUID = ds.getString(Tags.SeriesInstanceUID, "");           // Type 1
            // seriesNumber = ds.getString(Tags.SeriesNumber, "");                  // Type 2
            int i = ds.getInt(Tags.SeriesNumber, 1);                                // Type 2
            seriesNumber = String.valueOf(i);
            if (i < 1000) {
                seriesNumber = "000" + seriesNumber;
                seriesNumber =seriesNumber.substring(seriesNumber.length() - 3);
            }
        }
        
        
        public void addInstanceModule(Dataset ds) {
            
            // Image General Module, PS 3.3 - C.7.6.1 General Image Module
            // instanceNumber = ds.getString(Tags.InstanceNumber, "");              // Type 2
            int i = ds.getInt(Tags.InstanceNumber, 1);                              // Type 2
            instanceNumber = String.valueOf(i);
            if (i < 1000) {
                instanceNumber = "000" + instanceNumber;
                instanceNumber = instanceNumber.substring(instanceNumber.length() - 3);
            }
            
            // SOP Common Module, PS 3.3 - C.12.1, M
            sopClassUID = ds.getString(Tags.SOPClassUID, "");                       // Type 1
            sopInstanceUID = ds.getString(Tags.SOPInstanceUID, "");                 // Type 1
        }
        
        
        public void buildFilePath(String[] refFileIDs) {
            File f;
            
            // refFileIDs must be given
            if (refFileIDs == null) return;
            
            // Start with DICOMDIR directory
            f = dicomdir.getParentFile();
            for (int i = 0; i < refFileIDs.length; i++) {
                // Add descending IDs
                f = new File(f, refFileIDs[i]);
            }
            
            // Store the complete path in field
            filePath = f.getPath();
            // log.info(filePath);
        }
        
        
        public DcmObjectRecord(ResultSet rs) throws SQLException {
            
            recordID = rs.getInt("recordID");
            isProcessed = rs.getBoolean("isProcessed");
            
            filePath = rs.getString("filePath");
            
            patientBirthDate = rs.getString("patientBirthDate");
            patientName = rs.getString("patientName");
            patientID = rs.getString("patientID");
            patientBirthDate = rs.getString("patientBirthDate");
            patientSex = rs.getString("patientSex");
            studyInstanceUID = rs.getString("studyInstanceUID");
            studyDate = rs.getString("studyDate");
            studyTime = rs.getString("studyTime");
            referringPhysicianName = rs.getString("referringPhysicianName");
            studyID = rs.getString("studyID");
            accessionNumber = rs.getString("accessionNumber");
            studyDescription = rs.getString("studyDescription");
            modality = rs.getString("modality");
            seriesInstanceUID = rs.getString("seriesInstanceUID");
            seriesNumber = rs.getString("seriesNumber");
            instanceNumber = rs.getString("instanceNumber");
            sopClassUID = rs.getString("sopClassUID");
            sopInstanceUID = rs.getString("sopInstanceUID");
        }
        
        
        public String toPatientString() {
            String s  = patientName + ", ";
            s += formatDate(patientBirthDate) + ", ";
            s += patientSex + ", ";
            s += patientID;
            return s;
        }
        
        public String toStudyString() {
            String s  = "Study: " + studyID + ", ";
            s += formatDate(studyDate) + ", ";
            s += formatTime(studyTime) + ", ";
            s += studyDescription;
            return s;
        }
        
        public String toSeriesString() {
            String s  = "Series: " + seriesNumber + ", ";
            s += modality;
            return s;
        }
        
        public String toInstanceString() {
            String s  = "Instance: " + instanceNumber;
            return s;
        }
        
        private String formatDate(String date) {
            if (date.length() < 7) return date;
            
            String s = date.substring(0, 4) + "-";
            s += date.substring(4, 6) + "-";
            s += date.substring(6);
            return s;
        }
        
        private String formatTime(String time) {
            if (time.length() < 4) return time;
            
            String s = time.substring(0, 2) + ":";
            s += time.substring(2, 4);
            return s;
        }
    }
